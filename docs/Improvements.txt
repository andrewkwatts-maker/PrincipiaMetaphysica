from eml_math.operators import ops   # your existing ops module
from eml_math.pair import EMLPair

def boost(
    self,
    phi: Union[float, "EMLPoint"],
    c: float = 1.0,
    pure_eml: bool = False
) -> "EMLPoint":
    """Apply a Lorentz boost by rapidity φ. Returns new EMLPoint with identical Δ_M.

    φ can be a float (fast) or an EMLPoint (symbolic tree).
    Preserves Minkowski invariant exactly.
    """
    if isinstance(phi, EMLPoint):
        phi = phi.tension() if not pure_eml else phi  # extract or keep tree

    p = self.pair()                     # (t = exp(x), space = ln|y|)
    t = p.real_tension * c
    space = p.imag_tension

    if pure_eml:
        # Pure EML tree version — fully symbolic
        cosh_phi = ops.cosh(EMLPoint(phi, 1.0)).tension() if isinstance(phi, (int, float)) else ops.cosh(phi).tension()
        sinh_phi = ops.sinh(EMLPoint(phi, 1.0)).tension() if isinstance(phi, (int, float)) else ops.sinh(phi).tension()
    else:
        # Fast numeric (default)
        cosh_phi = math.cosh(phi)
        sinh_phi = math.sinh(phi)

    t_new = cosh_phi * t + sinh_phi * space
    space_new = sinh_phi * t + cosh_phi * space

    # Invert back to new EMLPoint(x', y')
    # x' = ln(t_new), y' = exp(space_new)
    if t_new <= 0:
        raise ValueError("Boost produced non-physical negative time component")
    x_new = math.log(t_new)
    y_new = math.exp(space_new)

    new_point = self.__class__(x_new, y_new)
    return new_point

import math
from dataclasses import dataclass
from typing import Callable, Optional
from eml_math.point import EMLPoint
from eml_math.momentum import FourMomentum

@dataclass
class MetricTensor:
    """Curved spacetime metric built from EMLPoint coordinates.
    g_mu_nu is supplied as a function of (x,y) that returns the 2x2 matrix.
    Defaults to Minkowski (+---) if none given.
    """
    g: Optional[Callable[[EMLPoint], list[list[float]]]] = None
    signature: str = "+---"

    def __post_init__(self):
        if self.g is None:
            # Default flat Minkowski
            self.g = lambda p: [[1.0, 0.0], [0.0, -1.0]]

    def ds2(self, p: EMLPoint, dx: float, dy: float) -> float:
        """Compute ds² = g_{\mu\nu} dx^μ dx^ν for infinitesimal displacement."""
        mat = self.g(p)
        t = p.pair().real_tension
        x = p.pair().imag_tension
        dt = dx * t   # scale by coordinate differentials (simplified 1+1D)
        dx_ = dy * x
        return mat[0][0]*dt**2 + 2*mat[0][1]*dt*dx_ + mat[1][1]*dx_**2

    def proper_time(self, p: EMLPoint, dx: float = 0.0, dy: float = 0.0) -> float:
        """√|ds²| — reduces to minkowski_delta() in flat space."""
        ds2 = self.ds2(p, dx, dy)
        return math.sqrt(max(abs(ds2), 0.0))

    def is_curved(self) -> bool:
        """Quick check: is this metric non-Minkowski?"""
        flat = self.g(EMLPoint(0,1))
        return abs(flat[0][0] - 1.0) > 1e-8 or abs(flat[1][1] + 1.0) > 1e-8

import numpy as np
from typing import Tuple

def christoffel(self, lam: int, mu: int, nu: int, p: EMLPoint) -> float:
    """Christoffel symbol Γ^λ_μν at point p (numeric, 1+1D)."""
    # Finite-difference approximation (accurate enough for integration)
    h = 1e-6
    g = lambda q: np.array(self.g(q))  # 2x2 metric matrix at q

    g0 = g(p)
    g_inv = np.linalg.inv(g0)

    dg_mu = (g(EMLPoint(p.x + h, p.y)) - g(EMLPoint(p.x - h, p.y))) / (2 * h)
    dg_nu = (g(EMLPoint(p.x, p.y + h)) - g(EMLPoint(p.x, p.y - h))) / (2 * h)  # y ~ r

    # Full sum (only 4 terms in 1+1D)
    term = 0.0
    for sigma in range(2):
        d1 = dg_mu[mu, sigma] if mu == 0 else dg_nu[mu, sigma]  # simplified coord scaling
        d2 = dg_nu[nu, sigma] if nu == 0 else dg_mu[nu, sigma]
        d3 = (g(EMLPoint(p.x + (1 if mu==0 else 0), p.y + (1 if mu==1 else 0))) -
              g(EMLPoint(p.x - (1 if mu==0 else 0), p.y - (1 if mu==1 else 0)))) / (2 * h)
        term += g_inv[lam, sigma] * (d1 + d2 - d3[mu, nu])
    return 0.5 * term

# Schwarzschild analytic override (exact, zero numerical error)
@staticmethod
def schwarzschild_christoffel(lam: int, mu: int, nu: int, r: float, rs: float = 2.0) -> float:
    """Exact analytic Christoffels for Schwarzschild 1+1D radial (from sympy derivation)."""
    if lam == 0 and mu == 0 and nu == 1:   # Γ^t_{t r}
        return 0.5 * rs / (r * (r - rs))
    if lam == 1 and mu == 0 and nu == 0:   # Γ^r_{t t}
        return 0.5 * rs * (r - rs) / r**3
    if lam == 1 and mu == 1 and nu == 1:   # Γ^r_{r r}
        return -0.5 * rs / (r * (r - rs))
    return 0.0  # all others zero for radial slice

def geodesic_step(
    self,
    metric: "MetricTensor",
    dtau: float = 0.01,
    use_analytic: bool = True
) -> "EMLState":
    """Advance one proper-time step along the geodesic using the full equation.
    Uses Christoffel symbols + current 4-velocity.
    """
    fm = FourMomentum(self.point)          # current 4-momentum
    p = self.point
    r = math.exp(p.x)                      # radial coord

    # 4-velocity components (u^μ = dx^μ / dτ)
    u_t = fm.energy / fm.mass if abs(fm.mass) > 1e-12 else 1.0
    u_r = fm.momentum / fm.mass if abs(fm.mass) > 1e-12 else 0.0

    # Christoffel terms
    if use_analytic and hasattr(metric, 'schwarzschild_christoffel'):
        Gamma_t_tr = metric.schwarzschild_christoffel(0, 0, 1, r)
        Gamma_r_tt = metric.schwarzschild_christoffel(1, 0, 0, r)
        Gamma_r_rr = metric.schwarzschild_christoffel(1, 1, 1, r)
    else:
        Gamma_t_tr = metric.christoffel(0, 0, 1, p)
        Gamma_r_tt = metric.christoffel(1, 0, 0, p)
        Gamma_r_rr = metric.christoffel(1, 1, 1, p)

    # Acceleration (geodesic equation right-hand side)
    d2t_dtau2 = -2 * Gamma_t_tr * u_t * u_r
    d2r_dtau2 = -Gamma_r_tt * u_t**2 - Gamma_r_rr * u_r**2

    # Simple Euler update for velocity then position (Verlet-style for stability)
    u_t_new = u_t + d2t_dtau2 * dtau
    u_r_new = u_r + d2r_dtau2 * dtau

    # New coordinates via integrated displacement
    dt = u_t_new * dtau
    dr = u_r_new * dtau
    t_new = math.exp(p.x) + dt
    r_new = r + dr

    # Re-encode as new EMLPoint (inverse mapping)
    x_new = math.log(t_new)
    y_new = math.exp(r_new) if r_new > 0 else 1.0  # safety

    new_point = EMLPoint(x_new, y_new)
    return EMLState.from_point(new_point)  # your existing factory

    Components: [real, e1, e2, e3, e4, e5, e6, e7]

    def __post_init__(self):
        if len(self.components) != 8:
            raise ValueError("Octonion requires exactly 8 EMLPoints")
        self.numeric = np.array([c.pair().real_tension for c in self.components])

    def __mul__(self, other: "Octonion") -> "Octonion":
        """Non-associative octonion multiplication using the standard table."""
        if not isinstance(other, Octonion):
            other = Octonion([other] + [EMLPoint(0,1)]*7)  # scalar promotion
        result = np.zeros(8)
        for i in range(8):
            for j in range(8):
                k = MULT_TABLE[i, j]
                sign = 1 if k >= 0 else -1
                k = abs(k)
                result[k] += sign * self.numeric[i] * other.numeric[j]
        # Re-encode as new EMLPoints (symbolic tree preserved in original)
        new_comps = [EMLPoint(math.log(abs(v) + 1e-12), 1.0) for v in result]
        return Octonion(new_comps)

        """Octonion conjugate: real stays, imaginaries flip sign."""
        conj_comps = [self.components[0]] + [-c for c in self.components[1:]]
        return Octonion(conj_comps)

        """|o| = √(o * conjugate(o)) — multiplicative and equals EMLNDVector.euclidean_norm."""
        o_bar = self.conjugate()
        prod = self * o_bar
        return math.sqrt(prod.numeric[0])  # real part only

        """Bridge to your existing N-dim lattice machinery."""
        return EMLNDVector(self.components)

        return f"Octonion({self.numeric.round(4)})"
